import pandas as pd
import polars as pl
import time

# Create a Pandas DataFrame
pandas_df = pd.DataFrame({'A': range(1, 1000001), 'B': range(1000001, 2000001)})

# Create a Polars DataFrame from the Pandas DataFrame
polars_df = pl.from_pandas(pandas_df)

# Perform a simple data manipulation operation using Pandas
start_time = time.time()
pandas_result = pandas_df['A'] + pandas_df['B']
pandas_execution_time = time.time() - start_time

# Perform the same data manipulation operation using Polars
start_time = time.time()
polars_result = polars_df['A'] + polars_df['B']
polars_execution_time = time.time() - start_time

# Print the results and execution times
print("Pandas result:")
print(pandas_result)
print("Execution time: ", pandas_execution_time, " seconds")

print("\nPolars result:")
print(polars_result)
print("Execution time: ", polars_execution_time, " seconds")
